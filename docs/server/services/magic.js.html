classify



used to generate an array of array that groups the t array by the atr field


function classify(t = [], atr) {
  var tmp = Array(...t)
  var out = []
  while (tmp.length) {
    let i = tmp[0]
    out[out.length] = tmp.filter(j => j[atr] == i[atr])
    tmp = tmp.filter(j => j[atr] != i[atr])
  }
  return out
}
